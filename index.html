public static void main(String[] args) {
    String input = "halloxhallo";
    String delete = "aox";
    String output = "";

    for (int i = 0; i < delete.length(); i++) {
        output = "";
        for (int j = 0; j < input.length(); j++) {

                counter++;
            }

     * checks, if the character "zeichen" is a vocal

     * @return
